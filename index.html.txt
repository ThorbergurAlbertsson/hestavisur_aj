<script>
    const visur = [
        {
            flokkur: "barnavísur",
            titill: "Lítill hestur",
            texti: "Lítill hestur hleypur hratt..."
        },
        {
            flokkur: "gamlar",
            titill: "Hestur á heiđi",
            texti: "Hestur stendur úti á heiđi..."
        },
        {
            flokkur: "nyjar",
            titill: "Ný vísa",
            texti: "Ţú setur ţína eigin vísu hér..."
        }
    ];

    function birtaVisur(filter = "") {
        const content = document.getElementById("content");
        content.innerHTML = "";

        visur
            .filter(v => v.texti.toLowerCase().includes(filter.toLowerCase()))
            .forEach(v => {
                const box = document.createElement("div");
                box.className = "visa";
                box.innerHTML = `<h2>${v.titill}</h2><p>${v.texti}</p>`;
                content.appendChild(box);
            });
    }

    document.getElementById("search").addEventListener("input", e => {
        birtaVisur(e.target.value);
    });

    birtaVisur();
</script>
